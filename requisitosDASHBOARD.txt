import { Users } from 'lucide-react';

export function QueueStatusCard() {
  const queueStatus = {
    currentQueue: 8,
    recentJoins: 3
  };

  const getStatusColor = () => {
    if (queueStatus.currentQueue <= 5) return 'bg-green-100 text-green-600 border-green-200';
    if (queueStatus.currentQueue <= 10) return 'bg-yellow-100 text-yellow-600 border-yellow-200';
    return 'bg-red-100 text-red-600 border-red-200';
  };

  const getStatusText = () => {
    if (queueStatus.currentQueue <= 5) return 'Fluxo Normal';
    if (queueStatus.currentQueue <= 10) return 'Moderado';
    return 'Fila Longa';
  };

  return (
    <div className={`rounded-xl shadow-sm border-2 p-6 hover:shadow-md transition-shadow ${getStatusColor()}`}>
      <div className="flex items-center gap-2 mb-3">
        <Users className="w-5 h-5" />
        <p className="text-sm">Status da Fila</p>
      </div>
      
      <div className="mb-3">
        <p className="text-3xl mb-1">{queueStatus.currentQueue}</p>
        <p className="text-sm opacity-80">{getStatusText()}</p>
      </div>

      <div className="pt-3 border-t border-current/20 text-xs opacity-75">
        +{queueStatus.recentJoins} entraram nos últimos 5min
      </div>
    </div>
  );
}

import { History, UserCheck, UserX, Clock } from 'lucide-react';

interface HistoryItem {
  id: number;
  name: string;
  action: 'completed' | 'abandoned';
  time: string;
  waitTime?: string;
}

const historyData: HistoryItem[] = [
  { id: 1, name: 'Julia Ferreira', action: 'completed', time: '28min atrás', waitTime: '8min' },
  { id: 2, name: 'Carlos Rodrigues', action: 'abandoned', time: '35min atrás', waitTime: '12min' },
  { id: 3, name: 'Ana Lima', action: 'completed', time: '42min atrás', waitTime: '5min' },
  { id: 4, name: 'Pedro Costa', action: 'completed', time: '58min atrás', waitTime: '7min' },
  { id: 5, name: 'Marcos Silva', action: 'abandoned', time: '1h atrás', waitTime: '15min' },
  { id: 6, name: 'Beatriz Santos', action: 'completed', time: '1h 15min atrás', waitTime: '6min' }
];

export function QueueHistory() {
  const completedCount = historyData.filter(h => h.action === 'completed').length;
  const abandonedCount = historyData.filter(h => h.action === 'abandoned').length;
  const abandonRate = ((abandonedCount / historyData.length) * 100).toFixed(0);

  return (
    <div className="bg-white rounded-xl shadow-sm border border-gray-200 p-6">
      <div className="mb-6">
        <h2 className="text-gray-900 flex items-center gap-2">
          <History className="w-5 h-5 text-indigo-600" />
          Histórico da Fila
        </h2>
        <p className="text-gray-600 text-sm mt-1">Últimas 24 horas</p>
      </div>

      {/* Stats */}
      <div className="grid grid-cols-2 gap-3 mb-4">
        <div className="bg-green-50 rounded-lg p-3 border border-green-200">
          <div className="flex items-center gap-2 mb-1">
            <UserCheck className="w-4 h-4 text-green-600" />
            <span className="text-green-900 text-sm">Atendidos</span>
          </div>
          <p className="text-green-900 text-xl">{completedCount}</p>
        </div>
        <div className="bg-red-50 rounded-lg p-3 border border-red-200">
          <div className="flex items-center gap-2 mb-1">
            <UserX className="w-4 h-4 text-red-600" />
            <span className="text-red-900 text-sm">Desistiram</span>
          </div>
          <p className="text-red-900 text-xl">{abandonedCount}</p>
        </div>
      </div>

      {/* History List */}
      <div className="space-y-2 max-h-[300px] overflow-y-auto">
        {historyData.map((item) => (
          <div
            key={item.id}
            className={`flex items-center gap-3 p-3 rounded-lg border transition-colors ${
              item.action === 'completed'
                ? 'bg-green-50 border-green-200 hover:bg-green-100'
                : 'bg-red-50 border-red-200 hover:bg-red-100'
            }`}
          >
            <div className={`p-2 rounded-lg shrink-0 ${
              item.action === 'completed' ? 'bg-green-100' : 'bg-red-100'
            }`}>
              {item.action === 'completed' ? (
                <UserCheck className="w-4 h-4 text-green-600" />
              ) : (
                <UserX className="w-4 h-4 text-red-600" />
              )}
            </div>

            <div className="flex-1 min-w-0">
              <p className={`truncate ${
                item.action === 'completed' ? 'text-green-900' : 'text-red-900'
              }`}>
                {item.name}
              </p>
              <div className="flex items-center gap-2 text-xs mt-1">
                {item.waitTime && (
                  <div className={`flex items-center gap-1 ${
                    item.action === 'completed' ? 'text-green-700' : 'text-red-700'
                  }`}>
                    <Clock className="w-3 h-3" />
                    <span>Aguardou {item.waitTime}</span>
                  </div>
                )}
                <span className={`${
                  item.action === 'completed' ? 'text-green-600' : 'text-red-600'
                }`}>
                  • {item.time}
                </span>
              </div>
            </div>
          </div>
        ))}
      </div>

      {/* Summary */}
      <div className="mt-4 pt-4 border-t border-gray-200">
        <div className="flex items-center justify-between text-sm">
          <span className="text-gray-600">Taxa de Desistência</span>
          <span className={`${parseInt(abandonRate) > 20 ? 'text-red-600' : 'text-green-600'}`}>
            {abandonRate}%
          </span>
        </div>
      </div>
    </div>
  );
}

import { Package, AlertTriangle, TrendingDown } from 'lucide-react';

interface StockItem {
  name: string;
  current: number;
  max: number;
  unit: string;
  weeklyConsumption: number;
  daysRemaining: number;
}

const stockItems: StockItem[] = [
  { name: 'Grãos de Café', current: 3.5, max: 10, unit: 'kg', weeklyConsumption: 2.1, daysRemaining: 12 },
  { name: 'Leite', current: 8, max: 20, unit: 'L', weeklyConsumption: 5.2, daysRemaining: 11 },
  { name: 'Açúcar', current: 1.2, max: 5, unit: 'kg', weeklyConsumption: 0.8, daysRemaining: 10 },
  { name: 'Copos Descartáveis', current: 150, max: 500, unit: 'un', weeklyConsumption: 98, daysRemaining: 11 }
];

export function StockControl() {
  const getStatusColor = (percentage: number) => {
    if (percentage <= 25) return 'text-red-600';
    if (percentage <= 50) return 'text-yellow-600';
    return 'text-green-600';
  };

  const getBarColor = (percentage: number) => {
    if (percentage <= 25) return 'bg-red-500';
    if (percentage <= 50) return 'bg-yellow-500';
    return 'bg-green-500';
  };

  return (
    <div className="bg-white rounded-xl shadow-sm border border-gray-200 p-6">
      <div className="mb-6">
        <h2 className="text-gray-900 flex items-center gap-2">
          <Package className="w-5 h-5 text-blue-600" />
          Controle de Estoque
        </h2>
        <p className="text-gray-600 text-sm mt-1">Níveis atuais e previsões</p>
      </div>

      <div className="space-y-4">
        {stockItems.map((item) => {
          const percentage = (item.current / item.max) * 100;
          const isLow = percentage <= 25;

          return (
            <div key={item.name} className={`p-4 rounded-lg border ${isLow ? 'bg-red-50 border-red-200' : 'bg-gray-50 border-gray-200'}`}>
              <div className="flex items-center justify-between mb-2">
                <div className="flex items-center gap-2">
                  <span className="text-gray-900">{item.name}</span>
                  {isLow && <AlertTriangle className="w-4 h-4 text-red-600" />}
                </div>
                <span className={`${getStatusColor(percentage)}`}>
                  {item.current} {item.unit}
                </span>
              </div>

              {/* Progress Bar */}
              <div className="mb-2">
                <div className="w-full bg-gray-200 rounded-full h-2 overflow-hidden">
                  <div
                    className={`h-full ${getBarColor(percentage)} transition-all rounded-full`}
                    style={{ width: `${percentage}%` }}
                  />
                </div>
                <div className="flex items-center justify-between mt-1 text-xs text-gray-600">
                  <span>{percentage.toFixed(0)}% do estoque</span>
                  <span>Máx: {item.max} {item.unit}</span>
                </div>
              </div>

              {/* Stats */}
              <div className="flex items-center justify-between text-xs text-gray-600">
                <div className="flex items-center gap-1">
                  <TrendingDown className="w-3 h-3" />
                  <span>{item.weeklyConsumption} {item.unit}/semana</span>
                </div>
                <span className={isLow ? 'text-red-600 font-medium' : ''}>
                  ~{item.daysRemaining} dias restantes
                </span>
              </div>
            </div>
          );
        })}
      </div>

      {/* Alert Summary */}
      <div className="mt-4 pt-4 border-t border-gray-200">
        <div className="bg-red-50 border border-red-200 rounded-lg p-3">
          <div className="flex items-start gap-2">
            <AlertTriangle className="w-4 h-4 text-red-600 mt-0.5 shrink-0" />
            <div>
              <p className="text-red-900 text-sm">Alerta de Estoque Baixo</p>
              <p className="text-red-700 text-xs mt-1">
                Grãos de Café e Açúcar precisam de reposição em breve
              </p>
            </div>
          </div>
        </div>
      </div>
    </div>
  );
}
